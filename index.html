// Under the Hood

Simple logic.
Powerful results.

MetricFlow is an "If This Then That" loop at its core. A Farcaster reaction triggers the agent, which validates the user, logs them to the allowlist, and fires an on-chain transaction — all in under 2 seconds.

Read the Source →
// rewardEngine.ts

async function processEvent(
  event: EngagementEvent
) {

  // Verify ETH address exists
  if (!event.reactorEthAddress)
    return skip("no_address");

  // Add to allowlist
  addToAllowlist(event);

  // Send ETH on Base
  const tx = await sendReward(
    event.reactorEthAddress,
    config.rewardAmountWei
  );

  return { status: "sent", tx };
}
